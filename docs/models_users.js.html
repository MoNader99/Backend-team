/**
 * Retrieves a user by token.
 * @method findByToken
 * @author ranime
 * @param {String} token - User token
 * @returns {User} User object
 * 
 */


UserSchema.statics.findByToken = function (token) {
    var User = this;
    var decoded;

    try {
        decoded = jwt.verify(token , 'secretkeyforuser');
    } catch (e) {
       return Promise.reject();
    }
    return User.findOne({
     _id:decoded._id
    });

};
